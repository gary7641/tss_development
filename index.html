function renderMfeMae(result) { const trades = result.trades; if (!trades || trades.length === 0) return; // 簡單計算 MFE / MAE (用現有 Max Profit / Max Loss) const dataPoints = trades.map((t, i) => ({ x: t.maxLoss || 0, // MAE (負數) y: t.maxProfit || 0, // MFE (正數) profit: t.profit, symbol: t.symbol, label: `Trade ${i+1}` })); // Scatter Plot const ctx = document.getElementById('mfeMaeChart').getContext('2d'); if (window.mfeChart) window.mfeChart.destroy(); window.mfeChart = new Chart(ctx, { type: 'scatter', data: { datasets: [{ label: 'Trades', data: dataPoints, backgroundColor: dataPoints.map(p => p.profit >= 0 ? '#4ade80' : '#f87171'), borderColor: '#1e2937', borderWidth: 1, pointRadius: 5, pointHoverRadius: 8 }] }, options: { responsive: true, maintainAspectRatio: false, plugins: { legend: { display: false }, tooltip: { callbacks: { label: (ctx) => `${ctx.raw.label} | Profit: ${ctx.raw.profit.toFixed(2)} | MFE: ${ctx.raw.y} | MAE: ${ctx.raw.x}` } } }, scales: { x: { title: { display: true, text: 'MAE (Max Adverse Excursion)' }, grid: { color: '#e2e8f0' } }, y: { title: { display: true, text: 'MFE (Max Favorable Excursion)' }, grid: { color: '#e2e8f0' } } } } }); // Summary const avgMFE = dataPoints.reduce((a, b) => a + b.y, 0) / dataPoints.length; const avgMAE = dataPoints.reduce((a, b) => a + b.x, 0) / dataPoints.length; document.getElementById('mfeMaeSummary').innerHTML = `
Avg MFE${avgMFE.toFixed(1)}
Avg MAE${avgMAE.toFixed(1)}
MFE/MAE Ratio${(avgMFE / Math.abs(avgMAE)).toFixed(2)}
`; // Top 5 const top = [...dataPoints].sort((a,b) => b.y - a.y).slice(0,5); let topHtml = ''; top.forEach(p => { topHtml += `${p.label} : MFE ${p.y.toFixed(1)} | MAE ${p.x.toFixed(1)}
`; }); topHtml += ''; document.getElementById('topMfeMae').innerHTML = topHtml; }
